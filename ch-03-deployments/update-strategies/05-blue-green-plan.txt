apiVersion: v1
kind: Namespace
metadata:
  name: blue-green
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: web-blue
  namespace: blue-green
data:
  index.html: |
    <html><body><h1>Hello from BLUE</h1></body></html>
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: web-blue
  namespace: blue-green
spec:
  replicas: 2
  selector:
    matchLabels:
      app: web
      color: blue
  template:
    metadata:
      labels:
        app: web
        color: blue
    spec:
      containers:
        - name: nginx
          image: nginx:stable-alpine
          ports:
          - containerPort: 80
          volumeMounts:
            - name: www
              mountPath: /usr/share/nginx/html/index.html
              subPath: index.html
      volumes:
        - name: www
          configMap:
            name: web-blue
---
